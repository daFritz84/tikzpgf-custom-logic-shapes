\usetikzlibrary{circuits.logic}
\usepgflibrary{shapes.gates.logic.custom}


%
% When this lib is loaded, all symbols default to IEC
%

\tikzset{
	circuit logic IEC/.style=
	{
		circuit logic,
		set halfsub graphic = halfsub graphic,
		set mux graphic = mux graphic,
		set dmux graphic = dmux graphic,
		set rslatch graphic = rslatch graphic,
		set dflipflop graphic = dflipflop graphic,
		%
		halfsub symbol = {$\scriptstyle HS$},
		mux symbol = {$\scriptstyle MUX$},
		dmux symbol = {$\scriptstyle DMUX$}
	},
}



\tikzset{
	%%circuit logic IEC make graphic=halfsub gate IEC,
	halfsub graphic/.style={
		circuit symbol open,
		circuit symbol size=width 6 height 6,
		shape=#1,
		inner sep=.5ex
	}
	mux graphic/.style={
		circuit symbol open,
		circuit symbol size=width 12 height 3,
		shape=#1,
		inner sep=.5ex
	}
	dmux graphic/.style={
		circuit symbol open,
		circuit symbol size=width 12 height 3,
		shape=#1,
		inner sep=.5ex
	}
	rslatch graphic/.style={
		circuit symbol open,
		circuit symbol size=width 1, height 1,
		shape=#1,
		inner sep=.5ex
	}
	dflipflop graphic/.style={
		circuit symbol open,
		circuit symbol size=width 1, height 1,
		shape=#1,
		inner sep=.5ex
	}
	jkflipflop graphic/.style={
		circuit symbol open,
		circuit symbol size=width 1, height 1,
		shape=#1,
		inner sep=.5ex
	}
}
